% ---- Helper function: pretty print positive and negative numbers
\ExplSyntaxOn
\NewExpandableDocumentCommand{\snum}{O{}m}
 {
  \mathbin
   {
    \fp_compare:nTF { #2 >= 0 }
     { \Plus }
     { \Minus }
   }
  \num[#1]{ \fp_eval:n { abs(#2) } }
 }
